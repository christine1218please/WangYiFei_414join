function bubbleSort(arr):
    n = length(arr)
    for i from 0 to n-2:
        for j from 0 to n-2-i:
            if arr[j] > arr[j+1]:
                swap(arr[j], arr[j+1])

function swap(a, b):
    temp = a
    a = b
    b = temp
